    def preparar_expresion(expr: str) -> str:
    expr = expr.replace(" ", "")
    expr = expr.replace("^", "**")
    expr = expr.replace("-x", "-1*x")
    expr = expr.replace("+x", "+1*x")
    if expr.startswith("x"):
        expr = "1*" + expr
    expr = expr.replace("x", "*x")
    expr = expr.replace("**x", "*x")
    return expr

    Python - Newton Raphson

    def f(x):
    return x**3 - x - 2

def df(x):
    return 3*x**2 - 1

def newton_raphson(x0, tolerancia=1e-6, max_iter=100):
    print("\nM√âTODO DE NEWTON-RAPHSON")
    print("Iter |     x      |    f(x)    |   Error")
    print("-" * 50)
    
    for i in range(1, max_iter + 1):
        fx = f(x0)
        dfx = df(x0)
        
        if dfx == 0:
            print("Error: derivada es cero.")
            return None
        
        x1 = x0 - fx / dfx
        error = abs(x1 - x0)
        
        print(f"{i:4d} | {x1:9.6f} | {fx:10.6f} | {error:9.6f}")
        
        if error < tolerancia:
            print("\nRa√≠z aproximada encontrada:", x1)
            return x1
        
        x0 = x1
    
    print("\nNo se alcanz√≥ la tolerancia.")
    return x1

if __name__ == "__main__":
    print("M√âTODO DE NEWTON-RAPHSON")

    print("Derivada: f'(x) = 3x^2 - 1\n")

    def calcular_indice_h(citas):
    """
    Calcula el √≠ndice H de un investigador.
    citas: lista de citas por publicaci√≥n
    """
    citas.sort(reverse=True)
    h = 0
    for i, c in enumerate(citas):
        if c >= i + 1:
            h = i + 1
        else:
            break
    return h

    def metodo_secante(x0, x1, tol=1e-6, max_iter=100):
    print("Iter |     x0     |     x1     |     x2     |   f(x2)")
    print("-" * 60)
    
    for i in range(max_iter):
        fx0 = f(x0)
        fx1 = f(x1)
        
        if fx1 - fx0 == 0:
            print("Divisi√≥n por cero")
            return None
        
        x2 = x1 - fx1 * (x1 - x0) / (fx1 - fx0)
        fx2 = f(x2)
        
        print(f"{i+1:4d} | {x0:9.5f} | {x1:9.5f} | {x2:9.5f} | {fx2:9.5f}")
        
        if abs(fx2) < tol:
            print(f"\nRa√≠z encontrada: {x2}")
            return x2
        
        x0, x1 = x1, x2
    
    print("M√°ximo de iteraciones alcanzado")
    return x2

    fa = f(a)
