def calcular_indice_h(publicaciones):
    """
    Calcula el √≠ndice h de un investigador
    publicaciones: lista de citas ordenada descendente
    """
    publicaciones.sort(reverse=True)
    h = 0
    for i, citas in enumerate(publicaciones):
        if citas >= i + 1:
            h = i + 1
        else:
            break
    return h

# Ejemplo para un docente
publicaciones = [45, 32, 21, 15, 8, 5, 3]
h_index = calcular_indice_h(publicaciones)
print(f"√çndice H del docente: {h_index}")

import numpy as np

def gradiente_descendente(X, y, theta, alpha, iteraciones):
    """
    X: matriz de caracter√≠sticas
    y: vector objetivo
    theta: par√°metros iniciales
    alpha: tasa de aprendizaje
    iteraciones: n√∫mero de iteraciones
 theta: par√°metros iniciales
    alpha: tasa de aprendizaje
    iteraciones: n√∫mero de iteraciones
    """
    m = len(y)
    costo_historial = []
    
    for i in range(iteraciones):
        # Calcular predicciones
        h = X.dot(theta)
        
        # Calcular error
        error = h - y
        
        # Calcular gradiente
        gradiente = (1/m) * X.T.dot(error)
        
        # Actualizar par√°metros
        theta = theta - alpha * gradiente
        
        # Calcular costo
        costo = (1/(2*m)) * np.sum(error**2)
        costo_historial.append(costo)
        
        # Mostrar progreso cada 100 iteraciones
        if i % 100 == 0:
            print(f"Iteraci√≥n {i}: Costo = {costo}")
    
    return theta, costo_historial

# Datos de ejemplo
X = np.array([[1, 1], [1, 2], [1, 3]])
y = np.array([1, 2, 3])

Actividad 2.2: Simulaci√≥n de Reparto de Dulces

Problema: Un colegio quiere repartir 500 dulces y 300 chupetines entre 5 aulas.

Simulaci√≥n de Reparto 
def repartir_dulces(total_dulces, total_chupetines, aulas):
    """
    Reparte dulces y chupetines entre aulas proporcionalmente
    """
    reparto = {}
    estudiantes_totales = sum(aulas.values())
    
    print("=== REPARTO DE DULCES Y CHUPETINES ===")
    print(f"Total dulces: {total_dulces}")
    print(f"Total chupetines: {total_chupetines}")
    print(f"Total estudiantes: {estudiantes_totales}")
    print("-" * 40)
    
    for aula, estudiantes in aulas.items():
        # Calcular proporci√≥n
        proporcion = estudiantes / estudiantes_totales
        
        # Asignar recursos proporcionalmente
        dulces_asignados = int(total_dulces * proporcion)
        chupetines_asignados = int(total_chupetines * proporcion)
        
        reparto[aula] = {
            'estudiantes': estudiantes,
            'dulces': dulces_asignados,
            'chupetines': chupetines_asignados,
            'por_estudiante': {
                'dulces': round(dulces_asignados / estudiantes, 2),
                'chupetines': round(chupetines_asignados / estudiantes, 2)
            }
        }
        
        print(f"{aula}: {estudiantes} estudiantes")
        print(f"  Dulces: {dulces_asignados} ({reparto[aula]['por_estudiante']['dulces']}/estudiante)")
        print(f"  Chupetines: {chupetines_asignados} ({reparto[aula]['por_estudiante']['chupetines']}/estudiante)")
        print()
    
    return reparto

# Datos de ejemplo
aulas = {'Aula 1': 30, 'Aula 2': 25, 'Aula 3': 35, 'Aula 4': 28, 'Aula 5': 32}
reparto = repartir_dulces(500, 300, aulas)
