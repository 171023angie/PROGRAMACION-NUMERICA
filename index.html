import re

NUM_REGEX = r'^[+-]?(\d+(\.\d*)?|\.\d+)$'

def pedir_numero(nombre):
    while True:
        s = input(f"Ingrese {nombre}: ").strip()
        if re.match(NUM_REGEX, s):
            return float(s)
        print("Entrada inv√°lida. Introduce un n√∫mero (ej: 3, -2, 4.5, -0.7)")

def f(x, y):
    return 2 * y + x

def main():
    print("Programa: calcula r = 2*y y resultado = r + x usando f(x,y) = 2*y + x")
    x = pedir_numero("x")
    y = pedir_numero("y")

    r = 2 * y
    resultado = r + x
    resultado_func = f(x, y)

    print("\nValores calculados:")
    print(f"  x = {x}")
    print(f"  y = {y}")
    print(f"  r = 2 * y = {r}")
    print(f"  r + x = {resultado}")
    print(f"  f(x, y) = 2*y + x = {resultado_func}")

if __name__ == "__main__":
    main()

Las restricciones son condiciones o limitaciones que debe cumplir un problema matem√°tico o de programaci√≥n lineal. Estas limitaciones establecen los valores posibles que pueden tomar las variables, y definen la regi√≥n factible donde se buscar√° la mejor soluci√≥n.

import math
import re

def preparar_funcion(expr):
    expr = expr.replace(" ", "")
    expr = re.sub(r"(\d)([a-zA-Z])", r"\1*\2", expr)
    return expr

def graficar_restricciones(restricciones, fronteras, ancho=80, alto=25, escala=2):
    plano = [[" " for _ in range(ancho)] for _ in range(alto)]
    for x in range(ancho):
        plano[alto//2][x] = "-"
    for y in range(alto):
        plano[y][ancho//2] = "|"
    plano[alto//2][ancho//2] = "+"
    simbolos = ["*", "#", "+", "x"]
    for idx, expr in enumerate(fronteras):
        expr = preparar_funcion(expr)
        simbolo = simbolos[idx % len(simbolos)]
        for col in range(ancho):
            x = (col - ancho//2) / escala
            try:
                y = eval(expr, {"x": x, "math": math})
                fila = int(alto//2 - y * escala)
                if 0 <= fila < alto:
                    plano[fila][col] = simbolo
            except Exception:
                pass
    for fila in plano:
        print("".join(fila))

Definici√≥n: Se utilizan m√©todos num√©ricos para encontrar los valores de x que satisfacen f(x)=0, especialmente cuando no es posible resolver la ecuaci√≥n de manera algebraica.


def f(x):
    return x**3 + x - 1

def df(x):
    return 3*x**2 + 1

def newton_raphson(x0, tol=1e-6, max_iter=100):
    xn = x0
    for n in range(max_iter):
        fxn = f(xn)
        dfxn = df(xn)
        if dfxn == 0:
            print("Derivada cero. No se puede continuar.")
            return None
        xn1 = xn - fxn/dfxn
        print(f"Iteraci√≥n {n+1}: x = {xn1}, f(x) = {f(xn1)}")
        if abs(xn1 - xn) < tol:
            print(f"\nLa ra√≠z aproximada es: {xn1}")
            return xn1
        xn = xn1
    print("No se alcanz√≥ la tolerancia requerida")
    return None

x0 = 0.5
newton_raphson(x0)
