üîπ M√©todo de Newton-Raphson

El m√©todo de Newton-Raphson es una t√©cnica num√©rica usada para encontrar ra√≠ces de ecuaciones no lineales. Se basa en aproximar la funci√≥n mediante una recta tangente y actualizar sucesivamente la estimaci√≥n hasta alcanzar la ra√≠z deseada.

import math

def newton_raphson(f, df, x0, tol, max_iter):
    print("M√©todo de Newton-Raphson")
    for i in range(max_iter):
        x1 = x0 - f(x0) / df(x0)
        print(f"Iteraci√≥n {i+1}: x = {x1:.6f}")
        if abs(x1 - x0) < tol:
            print(f"Ra√≠z aproximada: {x1:.6f}")
            return x1
        x0 = x1
    print("No se alcanz√≥ la convergencia.")
    return None

# Ejemplo:
f = lambda x: x**3 - 4*x + 1
df = lambda x: 3*x**2 - 4

newton_raphson(f, df, x0=1.5, tol=1e-5, max_iter=10)

def secante(f, x0, x1, tol, max_iter):
    print("M√©todo de la Secante")
    for i in range(max_iter):
        x2 = x1 - f(x1)*(x1 - x0)/(f(x1) - f(x0))
        print(f"Iteraci√≥n {i+1}: x = {x2:.6f}")
        if abs(x2 - x1) < tol:
            print(f"Ra√≠z aproximada: {x2:.6f}")
            return x2
        x0, x1 = x1, x2
    print("No se alcanz√≥ la convergencia.")
    return None

f = lambda x: x**3 - 4*x + 1
secante(f, 1, 2, 1e-5, 10)

def punto_fijo(g, x0, tol, max_iter):
    print("M√©todo de Punto Fijo")
    for i in range(max_iter):
        x1 = g(x0)
        print(f"Iteraci√≥n {i+1}: x = {x1:.6f}")
        if abs(x1 - x0) < tol:
            print(f"Ra√≠z aproximada: {x1:.6f}")
            return x1
        x0 = x1
    print("No se alcanz√≥ la convergencia.")
    return None

g = lambda x: (4*x - 1)**(1/3)
punto_fijo(g, 1.5, 1e-5, 10)

def regula_falsi(f, a, b, tol, max_iter):
    print("M√©todo de Regula Falsi")
    for i in range(max_iter):
        x = b - f(b)*(b - a)/(f(b) - f(a))
        print(f"Iteraci√≥n {i+1}: x = {x:.6f}")
        if abs(f(x)) < tol:
            print(f"Ra√≠z aproximada: {x:.6f}")
            return x
        if f(a)*f(x) < 0:
            b = x
        else:
            a = x
    print("No se alcanz√≥ la convergencia.")
    return None

f = lambda x: x**3 - 4*x + 1
regula_falsi(f, 0, 1, 1e-5, 10)
