import math
import re

def preparar_funcion(expr):
    # Quitar espacios
    expr = expr.replace(" ", "")
    # Insertar * donde falte (ej: 5x -> 5*x)
    expr = re.sub(r"(\d)([a-zA-Z])", r"\1*\2", expr)
    return expr

def graficar_restricciones(restricciones, fronteras, ancho=80, alto=25, escala=2):
    # Matriz vac√≠a
    plano = [[" " for _ in range(ancho)] for _ in range(alto)]

    # Dibujar ejes
    for x in range(ancho):
        plano[alto//2][x] = "-"  # eje X
    for y in range(alto):
        plano[y][ancho//2] = "|"  # eje Y
    plano[alto//2][ancho//2] = "+"  # origen

    # Sombrar regi√≥n factible
    for col in range(ancho):
        for fila in range(alto):
            x = (col - ancho//2) / escala
            y = (alto//2 - fila) / escala
            try:
                if all(eval(cond, {"x":x,"y":y,"math":math}) for cond in restricciones):
                    plano[fila][col] = "."
            except Exception:
                pass

    # Dibujar fronteras (rectas o verticales)
    simbolos = ["*", "#", "+", "x"]
    for idx, expr in enumerate(fronteras):
        expr = preparar_funcion(expr)
        simbolo = simbolos[idx % len(simbolos)]
        for col in range(ancho):
            x = (col - ancho//2) / escala
            try:
                y = eval(expr, {"x": x, "math": math})
                fila = int(alto//2 - y * escala)
                if 0 <= fila < alto:
                    plano[fila][col] = simbolo
            except Exception:
                # Interpretar como vertical x = constante

                    vx = float(expr)
                    col_v = int(round(vx * escala + ancho//2))
                    if 0 <= col_v < ancho:
                        for fila_v in range(alto):
                            plano[fila_v][col_v] = simbolo
                except Exception:
                    pass

    # Imprimir plano
    for fila in plano:

üåü TEMA: M√âTODOS NUM√âRICOS PARA ENCONTRAR RA√çCES

Definici√≥n: Se utilizan m√©todos num√©ricos para encontrar los valores de x que satisfacen f(x)=0, especialmente cuando no es posible resolver la ecuaci√≥n de manera algebraica. Estos m√©todos buscan aproximar la ra√≠z con la precisi√≥n deseada mediante iteraciones.

üîπ Clasificaci√≥n de M√©todos

M√©todos Cerrados (de intervalo):

Requieren un intervalo [a, b] donde la funci√≥n cambia de signo (f(a)f(b)<0). Garantizan la convergencia.

  • M√©todo de Bisecci√≥n: Divide el intervalo a la mitad y selecciona el subintervalo donde la funci√≥n cambia de signo. Se repite hasta alcanzar la precisi√≥n deseada. Ventaja: siempre converge. Desventaja: lento.
  • M√©todo de Regla Falsa (Falsa Posici√≥n): Aproxima la ra√≠z usando la l√≠nea recta que une los extremos del intervalo. M√°s r√°pido que la bisecci√≥n en muchos casos.

M√©todos Abiertos (sin necesidad de intervalo):

  • M√©todo de Newton-Raphson: F√≥rmula: x‚Çô‚Çä‚ÇÅ = x‚Çô - f(x‚Çô)/f'(x‚Çô). Requiere derivada de la funci√≥n. Ventaja: muy r√°pido si la aproximaci√≥n inicial es buena. Desventaja: puede fallar si la derivada es cero o la aproximaci√≥n inicial est√° lejos de la ra√≠z.
  • M√©todo de la Secante: Similar a Newton-Raphson pero no requiere la derivada; usa dos aproximaciones iniciales. F√≥rmula: x‚Çô‚Çä‚ÇÅ = x‚Çô - f(x‚Çô) * (x‚Çô - x‚Çô‚Çã‚ÇÅ)/(f(x‚Çô) - f(x‚Çô‚Çã‚ÇÅ)).
  • M√©todo de Punto Fijo: Se transforma f(x)=0 en x=g(x) y se usa x‚Çô‚Çä‚ÇÅ = g(x‚Çô). Converge si |g'(x)| < 1 cerca de la ra√≠z.
# üßÆ M√©todo de Newton-Raphson ## üìò Descripci√≥n El **m√©todo de Newton-Raphson** es un algoritmo **num√©rico iterativo** que permite encontrar **ra√≠ces reales** de una funci√≥n \( f(x) = 0 \). Se basa en aproximar la funci√≥n mediante su **tangente** y actualizar sucesivamente el valor de la ra√≠z. --- ## üß† Fundamento Te√≥rico A partir de la ecuaci√≥n de la tangente en el punto \( (x_n, f(x_n)) \): \[ y = f(x_n) + f'(x_n)(x - x_n) \] Si \( y = 0 \), entonces: \[ x_{n+1} = x_n - \frac{f(x_n)}{f'(x_n)} \] Esta f√≥rmula permite calcular una nueva aproximaci√≥n de la ra√≠z a partir de la anterior. --- ## ‚öôÔ∏è Procedimiento 1. Elegir una **aproximaci√≥n inicial** \( x_0 \). 2. Evaluar la funci√≥n \( f(x_n) \) y su derivada \( f'(x_n) \). 3. Calcular \( x_{n+1} \) con la f√≥rmula de Newton-Raphson. 4. Repetir el proceso hasta que el **error** sea menor a la tolerancia deseada. --- ## üíª Implementaci√≥n en Python ```python # M√©todo de Newton-Raphson para encontrar ra√≠ces de f(x) = 0 def f(x): return x**3 + x - 1 # Funci√≥n def df(x): return 3*x**2 + 1 # Derivada def newton_raphson(x0, tol=1e-6, max_iter=100): xn = x0 for n in range(max_iter): fxn = f(xn) dfxn = df(xn) if dfxn == 0: print("Derivada cero. #  CODIGO M√©todo de Newton-Raphson para encontrar ra√≠ces de f(x) = 0

def f(x):
    return x**3 + x - 1  # Funci√≥n a evaluar

def df(x):
    return 3*x**2 + 1    # Derivada de la funci√≥n

def newton_raphson(x0, tol=1e-6, max_iter=100):
    xn = x0
    for n in range(max_iter):
        fxn = f(xn)
        dfxn = df(xn)
        if dfxn == 0:
            print("Derivada cero. No se puede continuar.")
            return None
        xn1 = xn - fxn/dfxn
        print(f"Iteraci√≥n {n+1}: x = {xn1}, f(x) = {f(xn1)}")
        if abs(xn1 - xn) < tol:
            print(f"\nLa ra√≠z aproximada es: {xn1}")
            return xn1
        xn = xn1
    print("No se alcanz√≥ la tolerancia requerida")
    return None

# Aproximaci√≥n inicial
x0 = 0.5

# Ejecutar el m√©todo
newton_raphson(x0)
