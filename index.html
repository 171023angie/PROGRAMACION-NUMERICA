Actividad: Programa f(x,y) = 2*y + x

Este programa pide dos n√∫meros, calcula r = 2*y y finalmente f(x,y) = 2*y + x:


import re

NUM_REGEX = r'^[+-]?(\d+(\.\d*)?|\.\d+)$'

def pedir_numero(nombre):
    while True:
        s = input(f"Ingrese {nombre}: ").strip()
        if re.match(NUM_REGEX, s):
            return float(s)
        print("Entrada inv√°lida. Introduce un n√∫mero (ej: 3, -2, 4.5, -0.7)")

def f(x, y):
    return 2 * y + x

def main():
    print("Programa: calcula r = 2*y y resultado = r + x usando f(x,y) = 2*y + x")
    x = pedir_numero("x")
    y = pedir_numero("y")

    r = 2 * y
    resultado = r + x
    resultado_func = f(x, y)

    print("\nValores calculados:")
    print(f"  x = {x}")
    print(f"  y = {y}")
    print(f"  r = 2 * y = {r}")
    print(f"  r + x = {resultado}")
    print(f"  f(x, y) = 2*y + x = {resultado_func}")

if __name__ == "__main__":
    main()

Actividad: M√©todo Newton-Raphson

Este programa aproxima la ra√≠z de una funci√≥n f(x)=0 usando el m√©todo de Newton-Raphson:


import re

NUM_REGEX = r'^[+-]?(\d+(\.\d*)?|\.\d+)$'

def pedir_numero(nombre):
    while True:
        s = input(f"Ingrese {nombre}: ").strip()
        if re.match(NUM_REGEX, s):
            return float(s)
        print("Entrada inv√°lida. Introduce un n√∫mero v√°lido (ej: 2, -1.5)")

def f(x):
    # Ejemplo: f(x) = x^3 + x - 1
    return x**3 + x - 1

def df(x):
    # Derivada de f(x)
    return 3*x**2 + 1

def newton_raphson(x0, tol=1e-6, max_iter=100):
    x = x0
    for i in range(max_iter):
        fx = f(x)
        dfx = df(x)
        if dfx == 0:
            print("Derivada cero. No se puede continuar.")
            return None
        x_new = x - fx/dfx
        if abs(x_new - x) < tol:
            return x_new
        x = x_new
    print("No converge en el n√∫mero m√°ximo de iteraciones")
    return None

def main():
    print("M√©todo Newton-Raphson para f(x) = x^3 + x - 1")
    x0 = pedir_numero("aproximaci√≥n inicial x0")
    raiz = newton_raphson(x0)
    if raiz is not None:
        print(f"Ra√≠z aproximada: {raiz}")
        print(f"f(ra√≠z) = {f(raiz)}")

if __name__ == "__main__":
    main()
