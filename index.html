    # ========================================================= # M√âTODO DE LA REGLA FALSA (REGULA FALSI) # ========================================================= def f(x): """ Funci√≥n a evaluar. Modifica esta funci√≥n seg√∫n el ejercicio. Ejemplo: f(x) = x^3 - x - 2 """ return x**3 - x - 2 def regula_falsi(a, b, tolerancia=1e-6, max_iter=100): """ Implementaci√≥n del m√©todo de la Regla Falsa. a, b : extremos del intervalo tolerancia : error permitido max_iter : n√∫mero m√°ximo de iteraciones """ if f(a) * f(b) >= 0: print("Error: f(a) y f(b) deben tener signos opuestos.") return None print("\nM√âTODO DE LA REGLA FALSA") print("Iter | a | b | xr | f(xr)") print("-" * 55) for i in range(1, max_iter + 1): xr = b - (f(b) * (a - b)) / (f(a) - f(b)) fxr = f(xr) print(f"{i:4d} | {a:9.5f} | {b:9.5f} | {xr:9.5f} | {fxr:9.5f}") if abs(fxr) < tolerancia: print("\nRa√≠z aproximada encontrada:", xr) return xr if f(a) * fxr < 0: b = xr else: a = xr print("\nNo se alcanz√≥ la tolerancia en el n√∫mero m√°ximo de iteraciones.") return xr # ===================== PROGRAMA PRINCIPAL ===================== if __name__ == "__main__": print("M√âTODO DE LA REGLA FALSA (REGULA FALSI)") print("Funci√≥n usada: f(x) = x^3 - x - 2\n") a = float(input("Ingrese el valor de a: ")) b = float(input("Ingrese el valor de b: ")) regula_falsi(a, b)

    # ========================================================= # M√âTODO DE NEWTON ‚Äì RAPHSON # ========================================================= # CONCEPTO: # El m√©todo de Newton‚ÄìRaphson es un m√©todo num√©rico iterativo # utilizado para encontrar ra√≠ces de ecuaciones no lineales, # es decir, valores de x tales que f(x) = 0. # # El m√©todo se basa en la aproximaci√≥n lineal de la funci√≥n # mediante su recta tangente en un punto inicial x0. # La intersecci√≥n de dicha recta con el eje x se toma como # una nueva aproximaci√≥n de la ra√≠z. # # F√≥rmula iterativa: # x_{n+1} = x_n - f(x_n) / f'(x_n) # # El m√©todo converge r√°pidamente cuando el valor inicial # est√° cerca de la ra√≠z y la derivada no es cero. # ========================================================= def f(x): """ Funci√≥n f(x). Ejemplo: f(x) = x^3 - x - 2 """ return x**3 - x - 2 def df(x): """ Derivada de la funci√≥n f(x). f'(x) = 3x^2 - 1 """ return 3*x**2 - 1 def newton_raphson(x0, tolerancia=1e-6, max_iter=100): """ Implementaci√≥n del m√©todo de Newton‚ÄìRaphson. x0 : valor inicial tolerancia : error permitido max_iter : n√∫mero m√°ximo de iteraciones """ print("\nM√âTODO DE NEWTON‚ÄìRAPHSON") print("Iter | x | f(x) | Error") print("-" * 50) for i in range(1, max_iter + 1): fx = f(x0) dfx = df(x0) if dfx == 0: print("Error: la derivada es cero. El m√©todo no puede continuar.") return None x1 = x0 - fx / dfx error = abs(x1 - x0) print(f"{i:4d} | {x1:9.6f} | {fx:10.6f} | {error:9.6f}") if error < tolerancia: print("\nRa√≠z aproximada encontrada:", x1) return x1 x0 = x1 print("\nNo se alcanz√≥ la tolerancia en el n√∫mero m√°ximo de iteraciones.") return x1 # ===================== PROGRAMA PRINCIPAL ===================== if __name__ == "__main__": print("M√âTODO DE NEWTON‚ÄìRAPHSON") print("Funci√≥n: f(x) = x^3 - x - 2") print("Derivada: f'(x) = 3x^2 - 1\n") x0 = float(input("Ingrese el valor inicial x0: ")) newton_raphson(x0)
