Este programa pide dos n√∫meros, calcula r = 2*y y finalmente f(x,y) = 2*y + x:


import re

NUM_REGEX = r'^[+-]?(\d+(\.\d*)?|\.\d+)$'

def pedir_numero(nombre):
    while True:
        s = input(f"Ingrese {nombre}: ").strip()
        if re.match(NUM_REGEX, s):
            return float(s)
        print("Entrada inv√°lida. Introduce un n√∫mero (ej: 3, -2, 4.5, -0.7)")

def f(x, y):
    return 2 * y + x

def main():
    print("Programa: calcula r = 2*y y resultado = r + x usando f(x,y) = 2*y + x")
    x = pedir_numero("x")
    y = pedir_numero("y")

    r = 2 * y
    resultado = r + x
    resultado_func = f(x, y)

    print("\nValores calculados:")
    print(f"  x = {x}")
    print(f"  y = {y}")
    print(f"  r = 2 * y = {r}")
    print(f"  r + x = {resultado}")
    print(f"  f(x, y) = 2*y + x = {resultado_func}")

if __name__ == "__main__":
