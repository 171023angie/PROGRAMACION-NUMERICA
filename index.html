TEMA: VARIABLES

La programación numérica es una rama de la programación que busca resolver problemas matemáticos mediante aproximaciones, usando algoritmos que permiten trabajar con cálculos complejos cuando no es posible obtener una solución exacta.

Actividad: Programa f(x,y) = 2*y + x

import re

NUM_REGEX = r'^[+-]?(\d+(\.\d*)?|\.\d+)$'

def pedir_numero(nombre):
    while True:
        s = input(f"Ingrese {nombre}: ").strip()
        if re.match(NUM_REGEX, s):
            return float(s)
        print("Entrada inválida. Introduce un número (ej: 3, -2, 4.5, -0.7)")

def f(x, y):
    return 2 * y + x

def main():
    print("Programa: calcula r = 2*y y resultado = r + x usando f(x,y) = 2*y + x")
    x = pedir_numero("x")
    y = pedir_numero("y")

    r = 2 * y
    resultado = r + x
    resultado_func = f(x, y)

    print("\nValores calculados:")
    print(f"  x = {x}")
    print(f"  y = {y}")
    print(f"  r = 2 * y = {r}")
    print(f"  r + x = {resultado}")
    print(f"  f(x, y) = 2*y + x = {resultado_func}")

if __name__ == "__main__":
    main()
TEMA: RESTRICCIONES

Las restricciones son condiciones o limitaciones que debe cumplir un problema matemático o de programación lineal. Estas limitaciones establecen los valores posibles que pueden tomar las variables, y definen la región factible donde se buscará la mejor solución.

Actividad: Graficar restricciones

import math
import re

def preparar_funcion(expr):
    expr = expr.replace(" ", "")
    expr = re.sub(r"(\d)([a-zA-Z])", r"\1*\2", expr)
    return expr

def graficar_restricciones(restricciones, fronteras, ancho=80, alto=25, escala=2):
    plano = [[" " for _ in range(ancho)] for _ in range(alto)]
    for x in range(ancho):
        plano[alto//2][x] = "-"
    for y in range(alto):
        plano[y][ancho//2] = "|"
    plano[alto//2][ancho//2] = "+"
    simbolos = ["*", "#", "+", "x"]
    for idx, expr in enumerate(fronteras):
        expr = preparar_funcion(expr)
        simbolo = simbolos[idx % len(simbolos)]
        for col in range(ancho):
            x = (col - ancho//2) / escala
            try:
                y = eval(expr, {"x": x, "math": math})
                fila = int(alto//2 - y * escala)
                if 0 <= fila < alto:
                    plano[fila][col] = simbolo
            except Exception:
                try:
                    vx = float(expr)
                    col_v = int(round(vx * escala + ancho//2))
                    if 0 <= col_v < ancho:
                        for fila_v in range(alto):
                            plano[fila_v][col_v] = simbolo
                except Exception:
                    pass
    for fila in plano:
        print("".join(fila))
